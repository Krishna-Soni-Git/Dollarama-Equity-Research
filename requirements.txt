# ── Dashboard framework ───────────────────────────────────────────────────────
# Streamlit handles the web UI, sidebar navigation, sliders, widgets,
# session state, and all st.* calls throughout Dashboard.py.
streamlit>=1.40.0,<2.0.0

# ── Interactive charts ────────────────────────────────────────────────────────
# Plotly powers every chart in the dashboard — bar charts, scatter plots,
# heatmaps, histograms, CDF curves. make_subplots is imported directly.
# NOTE: matplotlib is NOT used or required by this dashboard.
plotly>=5.20.0,<7.0.0

# ── Data manipulation ─────────────────────────────────────────────────────────
# Pandas handles all financial data — loading, transforming, styling tables,
# and passing DataFrames to Plotly and Streamlit.
pandas>=2.2.0

# ── Numerical computing + Monte Carlo simulation ──────────────────────────────
# NumPy is used for array operations, CAGR calculations, DCF projections,
# and the full 5,000-path Monte Carlo simulation in Tab 10.
numpy>=1.26.0

# ── Live financial data ───────────────────────────────────────────────────────
# yfinance fetches live DOL.TO data: annual financials (FY2022–FY2025),
# current price, valuation multiples, beta, 5-year daily price history,
# and full IPO-to-date price history from Oct 2009.
# Peer data (DLTR, DG) uses hardcoded SEC 10-K values — not live yfinance.
#
# PINNED BELOW 1.0.0 — yfinance 1.x introduced curl_cffi as a mandatory
# dependency which requires a C compiler to install. This fails silently on
# Windows without Visual Studio Build Tools and on macOS without Xcode
# Command Line Tools. Version 0.2.x installs cleanly on all platforms with
# no compilation step.
yfinance>=0.2.50,<1.0.0

# ── Machine learning — Tab 11 (Tutorial 5) ───────────────────────────────────
# scikit-learn provides LinearRegression, mean_squared_error, and r2_score
# for the ML Price Model tab. Imported inside a try/except so the rest of
# the dashboard still runs if scikit-learn is missing (Tab 11 shows an error).
scikit-learn>=1.3.0
